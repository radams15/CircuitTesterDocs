CLASS Circuit

	batteries: list<Element> -- all batteries in circuit
	resistors: list<Element> -- all resistors (and derivatives) in circuit
	current_sources: list<Element> -- all current sources in circuit
	elements: list<Element> -- all elements in circuit
	
	node_set: map<Int, Int> -- the nodes that was input, and the node that is assigned as the circuit
	node_count: Int -- the number of nodes
	nodes: list<Int> -- all the nodes

	FUNCTION init(circuit_elements)
		FOR e IN circuit_elements DO -- splits the elements array
			IF e IS Battery THEN
				batteries.append(e)
			ELSEIF e IS Resistor THEN
				resistors.append(e)
			ELSEIF e IS CurrentSource THEN
				current_sources.append(e)
			END IF
			
		END
		
		elements = batteries + resistors + current_sources  
		{- adds together all 3 arrays. Not using the function parameter
		as this could contain values that are not batteries,
		resistors or current sources -}
		
		FOR e IN elements DO
			node_set[e.n0] = e.n0
			node_set[e.n1] = e.n1
		END
		
		node_count = node_set.length()
		
		nodes = node_set.values()
	END FUNCTION
END
